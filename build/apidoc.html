function gulp-htmlone (opt)

  • description and source-code
    gulp-htmlone = function (opt) {

    var options = extend({
        removeSelector: '[will-remove]',
        keepliveSelector: '[keeplive]',
        destDir: './',
        coimport: true,
        cssminify: true,
        jsminify: true,
        noAlicdn: false
    }, (opt || {}));
    options.domain = options.publish ? linkHost.online : linkHost.daily;
    options.localError = false;
    if (options.noAlicdn) {
        options.domain = options.publish ? options.onlinePath : options.dailyPath;
        if (!options.domain) {
            options.localError = true;
        }
    } else {
        if (!options.appName || !options.appVersion) {
            options.localError = true;
        }
    }
    var _todo = 0;
    var _done = 0;

    function transform(file, enc, cb) {
        if (file.isNull()) return cb(null, file);
        if (file.isStream()) return cb(new PluginError(pluginName, 'Streaming not supported'));

        var data;
        var str = file.contents.toString('utf8');
        var filepath = file.path;

        _todo++;
        dealScripts(filepath, str, options, function(html) {
            var html = dealOtherLocalPath(html,options);
            file.contents = new Buffer(html);
            cb(null, file);

            _done++;
            if (_done === _todo) {
                gutil.log(gutil.colors.cyan('>> All html done!'));
            }
        });
    }

    return through.obj(transform);
}
